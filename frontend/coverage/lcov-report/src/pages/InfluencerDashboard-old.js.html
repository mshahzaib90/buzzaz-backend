                              <i className="bi bi-lock-fill me-1"></i>Private Account
                            </Badge>
                          ) : (
                            <Badge bg="success" className="me-2">
                              <i className="bi bi-unlock-fill me-1"></i>Public Account
                            </Badge>
                          )}
                          
                          {latestStats.isVerified ? (
                            <Badge bg="primary">
                              <i className="bi bi-patch-check-fill me-1"></i>Verified
                            </Badge>
                          ) : (
                            <Badge bg="light" text="dark">
                              <i className="bi bi-patch-exclamation me-1"></i>Not Verified
                            </Badge>
                          )}
                        </div>
                        <p className="small text-muted mb-3">{profile.location}</p>
                        
                        {/* Biography */}
                        {profile.bio && (
                          <div className="mb-3">
                            <p className="small text-start" style={{ fontSize: '0.9rem', lineHeight: '1.4' }}>
                              {profile.bio}
                            </p>
                          </div>
                        )}
 
                        <div className="d-flex justify-content-around text-center mt-3">
                          <div>
                            <div className="fw-bold text-primary">
                              {latestStats.followers?.toLocaleString() || 0}
                            </div>
                            <small className="text-muted">Followers</small>
                          </div>
                          <div>
                            <div className="fw-bold text-primary">
                              {latestStats.postsCount?.toLocaleString() || 0}
                            </div>
                            <small className="text-muted">Posts</small>
                          </div>
                          <div>
                            <div className="fw-bold text-primary">
                              {latestStats.following?.toLocaleString() || 0}
                            </div>
                            <small className="text-muted">Following</small>
                          </div>
                        </div>
                        <div className="mt-3">
                          <Button 
                            variant="outline-primary" 
                            size="sm"
                            onClick={handleSyncInstagram}
                            disabled={isUpdating}
                          >
                            {isUpdating ? (
                              <>
                                <Spinner as="span" animation="border" size="sm" className="me-2" />
                                Syncing...
                              </>
                            ) : (
                              <>
                                <i className="bi bi-arrow-repeat me-2"></i>
                                Sync Instagram Now
                              </>
                            )}
                          </Button>
                        </div>
                      </Card.Body>
                    </Card>
                  </Col>
 
                  <Col lg={8}>
                    <Card>
                      <Card.Header className="d-flex justify-content-between align-items-center">
                        <h5 className="mb-0">Profile Information</h5>
                        {!editMode ? (
                          <Button 
                            variant="outline-primary" 
                            size="sm"
                            onClick={() => setEditMode(true)}
                          >
                            <i className="bi bi-pencil me-1"></i>
                            Edit
                          </Button>
                        ) : (
                          <div>
                            <Button 
                              variant="outline-secondary" 
                              size="sm" 
                              className="me-2"
                              onClick={() => setEditMode(false)}
                            >
                              Cancel
                            </Button>
                            <Button 
                              variant="primary" 
                              size="sm"
                              onClick={handleUpdateProfile}
                              disabled={isUpdating}
                            >
                              {isUpdating ? (
                                <>
                                  <Spinner size="sm" className="me-1" />
                                  Saving...
                                </>
                              ) : (
                                'Save'
                              )}
                            </Button>
                          </div>
                        )}
                      </Card.Header>
                      <Card.Body>
                        {!editMode ? (
                          <div>
                            <div className="mb-3">
                              <strong>Bio:</strong>
                              <p className="mt-1 text-muted">{profile.bio}</p>
                            </div>
                            
                            <div className="mb-3">
                              <strong>Categories:</strong>
                              <div className="mt-1">
                                {profile.categories?.map((category, index) => (
                                  <Badge key={index} bg="light" text="dark" className="me-1 mb-1">
                                    {category}
                                  </Badge>
                                ))}
                              </div>
                            </div>
                            
                            <div className="mb-3">
                              <strong>Content Types:</strong>
                              <div className="mt-1">
                                {profile.contentTypes?.map((type, index) => (
                                  <Badge key={index} bg="light" text="dark" className="me-1 mb-1">
                                    {type}
                                  </Badge>
                                ))}
                              </div>
                            </div>
                          </div>
                        ) : (
                          <Form>
                            <Form.Group className="mb-3">
                              <Form.Label>Full Name</Form.Label>
                              <Form.Control
                                type="text"
                                value={editData.fullName}
                                onChange={(e) => setEditData({...editData, fullName: e.target.value})}
                              />
                            </Form.Group>
 
                            <Form.Group className="mb-3">
                              <Form.Label>Bio</Form.Label>
                              <Form.Control
                                as="textarea"
                                rows={3}
                                value={editData.bio}
                                onChange={(e) => setEditData({...editData, bio: e.target.value})}
                              />
                            </Form.Group>
 
                            <Form.Group className="mb-3">
                              <Form.Label>Location</Form.Label>
                              <Form.Control
                                type="text"
                                value={editData.location}
                                onChange={(e) => setEditData({...editData, location: e.target.value})}
                              />
                            </Form.Group>
 
                            <Form.Group className="mb-3">
                              <MultiSelect
                                label="Categories"
                                options={categoryOptions}
                                value={editData.categories}
                                onChange={(value) => setEditData({...editData, categories: value})}
                              />
                            </Form.Group>
 
                            <Form.Group className="mb-3">
                              <MultiSelect
                                label="Content Types"
                                options={contentTypeOptions}
                                value={editData.contentTypes}
                                onChange={(value) => setEditData({...editData, contentTypes: value})}
                              />
                            </Form.Group>
                          </Form>
                        )}
                      </Card.Body>
                    </Card>
                  </Col>
                </Row>
              </Tab.Pane>
 
              {/* Insights Tab */}
              <Tab.Pane eventKey="insights">
                <Row className="mb-4">
                  <Col lg={12}>
                    <Card>
                      <Card.Header className="d-flex justify-content-between align-items-center">
                          <h5 className="mb-0">Instagram Analytics</h5>
                          <div className="d-flex align-items-center gap-2">
                            <Button 
                              variant="outline-primary" 
                              size="sm" 
                              onClick={refreshData}
                              disabled={isLoading}
                            >
                              <i className={`bi bi-arrow-clockwise ${isLoading ? 'spin' : ''}`}></i>
                              {isLoading ? ' Refreshing...' : ' Refresh'}
                            </Button>
                            {profile?.instagramUsername && (
                              <a 
                                href={`https://instagram.com/${profile.instagramUsername}`}
                                target="_blank" 
                                rel="noopener noreferrer"
                                className="small"
                              >
                                View Profile
                              </a>
                            )}
                          </div>
                        </Card.Header>
                      <Card.Body>
                        <Row>
                          <Col md={4}>
                            <Card className="text-center h-100">
                              <Card.Body>
                                <i className="bi bi-people-fill text-primary display-6"></i>
                                <h4 className="mt-2 mb-1">{latestStats.followers?.toLocaleString() || 0}</h4>
                                <p className="text-muted mb-0">Followers</p>
                              </Card.Body>
                            </Card>
                          </Col>
                          <Col md={4}>
                            <Card className="text-center h-100">
                              <Card.Body>
                                <i className="bi bi-grid-3x3-gap-fill text-success display-6"></i>
                                <h4 className="mt-2 mb-1">{latestStats.postsCount?.toLocaleString() || 0}</h4>
                                <p className="text-muted mb-0">Posts</p>
                              </Card.Body>
                            </Card>
                          </Col>
                          <Col md={4}>
                            <Card className="text-center h-100">
                              <Card.Body>
                                <i className="bi bi-person-plus-fill text-info display-6"></i>
                                <h4 className="mt-2 mb-1">{latestStats.following?.toLocaleString() || 0}</h4>
                                <p className="text-muted mb-0">Following</p>
                              </Card.Body>
                            </Card>
                          </Col>
                        </Row>
                        <div className="mt-4">
                          <h6 className="mb-3">Recent Posts Engagement</h6>
                          {postsError && (
                            <Alert variant="danger" className="mb-3">{postsError}</Alert>
                          )}
                          {postsLoading ? (
                            <div className="d-flex align-items-center"><Spinner animation="border" size="sm" className="me-2" /> Loading posts</div>
                          ) : postMetrics.length > 0 ? (
                            <>
                              <ResponsiveContainer width="100%" height={300}>
                                <BarChart data={postMetrics} barCategoryGap="20%">
                                  <CartesianGrid strokeDasharray="3 3" />
                                  <XAxis dataKey="label" />
                                  <YAxis tickFormatter={(v) => `${Math.round(v)}%`} />
                                  <Tooltip formatter={(value, name) => name === 'engagement' ? [`${Number(value).toFixed(2)}%`, 'Engagement'] : [value.toLocaleString?.() ?? value, name]} />
                                  <Bar dataKey="engagement" name="Engagement %" fill="#0d6efd" />
                                </BarChart>
                              </ResponsiveContainer>
                              <Table bordered hover responsive size="sm" className="mt-3">
                                <thead>
                                  <tr>
                                    <th>Likes</th>
                                    <th>Comments</th>
                                    <th>Engagement %</th>
                                  </tr>
                                </thead>
                                <tbody>
                                  {postMetrics.map((r, i) => (
                                    <tr key={`${r.id}-${i}`}>
                                      <td>{r.likes.toLocaleString()}</td>
                                      <td>{r.comments.toLocaleString()}</td>
                                      <td>{r.engagement.toFixed(2)}%</td>
                                    </tr>
                                  ))}
                                </tbody>
                              </Table>
                              <div className="mt-3">
                                <p className="mb-1"><strong>Best performing post:</strong> {bestPost ? `${bestPost.engagement.toFixed(2)}%` : 'N/A'}</p>
                                <p className="mb-1"><strong>Lowest performing post:</strong> {worstPost ? `${worstPost.engagement.toFixed(2)}%` : 'N/A'}</p>
                                <p className="mb-0"><strong>Overall average engagement:</strong> {avgEngagement.toFixed(2)}%</p>
                              </div>
                            </>
                          ) : (
                            <p className="text-muted mb-0">No recent post data available.</p>
                          )}
                        </div>
                      </Card.Body>
                    </Card>
                  </Col>
                </Row>
 
                {/* YouTube Analytics */}
                {profile?.youtubeChannelId && (
                  <Row className="mb-4">
                    <Col lg={12}>
                      <Card>
                        <Card.Header className="d-flex justify-content-between align-items-center">
                           <h5 className="mb-0">YouTube Analytics</h5>
                           <div className="d-flex align-items-center gap-2">
                             <Button 
                               variant="outline-danger" 
                               size="sm" 
                               onClick={refreshData}
                               disabled={isLoading}
                             >
                               <i className={`bi bi-arrow-clockwise ${isLoading ? 'spin' : ''}`}></i>
                               {isLoading ? ' Refreshing...' : ' Refresh'}
                             </Button>
                             {youtubeStats?.url && (
                               <a href={youtubeStats.url} target="_blank" rel="noopener noreferrer" className="small">
                                 View Channel
                               </a>
                             )}
                           </div>
                         </Card.Header>
                        <Card.Body>
                          {youtubeError && (
                            <Alert variant="danger" className="mb-3">{youtubeError}</Alert>
                          )}
                          <Row>
                            <Col md={4} className="mb-3">
                              <Card className="text-center h-100">
                                <Card.Body>
                                  <i className="bi bi-youtube text-danger display-6"></i>
                                  <h6 className="mt-2 mb-1">Subscribers</h6>
                                  {isLoadingYT ? (
                                    <Spinner animation="border" size="sm" />
                                  ) : (
                                    <h4 className="mb-0">{(youtubeStats?.subscribers || 0).toLocaleString()}</h4>
                                  )}
                                </Card.Body>
                              </Card>
                            </Col>
                            <Col md={4} className="mb-3">
                              <Card className="text-center h-100">
                                <Card.Body>
                                  <i className="bi bi-camera-video-fill text-primary display-6"></i>
                                  <h6 className="mt-2 mb-1">Videos</h6>
                                  {isLoadingYT ? (
                                    <Spinner animation="border" size="sm" />
                                  ) : (
                                    <h4 className="mb-0">{(youtubeStats?.videos || 0).toLocaleString()}</h4>
                                  )}
                                </Card.Body>
                              </Card>
                            </Col>
                            <Col md={4} className="mb-3">
                              <Card className="text-center h-100">
                                <Card.Body>
                                  <i className="bi bi-bar-chart-fill text-success display-6"></i>
                                  <h6 className="mt-2 mb-1">Views</h6>
                                  {isLoadingYT ? (
                                    <Spinner animation="border" size="sm" />
                                  ) : (
                                    <h4 className="mb-0">{(youtubeStats?.views || 0).toLocaleString()}</h4>
                                  )}
                                </Card.Body>
                              </Card>
                            </Col>
                          </Row>
                          {/* Advanced metrics removed per request */}
                        </Card.Body>
                      </Card>
                    </Col>
                  </Row>
                )}
 
                {/* Growth Analytics removed per request */}
              </Tab.Pane>
 
              {/* YouTube Tab */}
              <Tab.Pane eventKey="youtube">
                {profile?.youtubeChannelId ? (
                  <Row className="mb-4">
                    <Col lg={12}>
                      <Card>
                        <Card.Header className="d-flex justify-content-between align-items-center">
                          <h5 className="mb-0">YouTube Analytics</h5>
                          {youtubeStats?.url && (
                            <a href={youtubeStats.url} target="_blank" rel="noopener noreferrer" className="small">
                              View Channel
                            </a>
                          )}
                        </Card.Header>
                        <Card.Body>
                          {youtubeError && (
                            <Alert variant="danger" className="mb-3">{youtubeError}</Alert>
                          )}
                          <Row>
                            <Col md={4}>
                              <Card className="text-center h-100">
                                <Card.Body>
                                  <i className="bi bi-youtube text-danger display-6"></i>
                                  <h6 className="mt-2 mb-1">Subscribers</h6>
                                  {isLoadingYT ? (
                                    <Spinner animation="border" size="sm" />
                                  ) : (
                                    <h4 className="mb-0">{(youtubeStats?.subscribers || 0).toLocaleString()}</h4>
                                  )}
                                </Card.Body>
                              </Card>
                            </Col>
                            <Col md={4}>
                              <Card className="text-center h-100">
                                <Card.Body>
                                  <i className="bi bi-play-circle text-primary display-6"></i>
                                  <h6 className="mt-2 mb-1">Videos</h6>
                                  {isLoadingYT ? (
                                    <Spinner animation="border" size="sm" />
                                  ) : (
                                    <h4 className="mb-0">{(youtubeStats?.videos || 0).toLocaleString()}</h4>
                                  )}
                                </Card.Body>
                              </Card>
                            </Col>
                            <Col md={4}>
                              <Card className="text-center h-100">
                                <Card.Body>
                                  <i className="bi bi-eye text-success display-6"></i>
                                  <h6 className="mt-2 mb-1">Total Views</h6>
                                  {isLoadingYT ? (
                                    <Spinner animation="border" size="sm" />
                                  ) : (
                                    <h4 className="mb-0">{(youtubeStats?.views || 0).toLocaleString()}</h4>
                                  )}
                                </Card.Body>
                              </Card>
                            </Col>
                          </Row>
                        </Card.Body>
                      </Card>
                    </Col>
                  </Row>
                ) : (
                  <Row className="mb-4">
                    <Col lg={12}>
                      <Card>
                        <Card.Body className="text-center py-5">
                          <i className="bi bi-youtube display-4 text-muted mb-3"></i>
                          <h6 className="text-muted">YouTube Not Connected</h6>
                          <p className="text-muted mb-3">Connect your YouTube channel to view real-time analytics</p>
                          <Button variant="danger" href="/connect-social">
                            <i className="bi bi-youtube me-2"></i>
                            Connect YouTube
                          </Button>
                        </Card.Body>
                      </Card>
                    </Col>
                  </Row>
                )}
              </Tab.Pane>
 
              {/* Chat Tab */}
              <Tab.Pane eventKey="chat">
                <ChatInterface currentUser={user} />
              </Tab.Pane>
            </Tab.Content>
          </Tab.Container>
        </Col>
      </Row>
    </Container>
  );
};
